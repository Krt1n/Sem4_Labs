1. Create Employee table with following constraints:  Make EmpNo as Primary key. Do not allow EmpName, Gender, Salary and Address to have null values. Allow Gender to have one of the two values: ‘M’, ‘F’.  

CREATE table Employee(EmpNo int PRIMARY KEY, EmpName varchar(20) NOT NULL, Gender char(1) NOT NULL CHECK(gender='M' or gender='F'), Salary float NOT NULL, Address varchar(20) NOT NULL, DNo int NOT NULL);

2. Create Department table with following: DeptNo as Primary Key. DeptName as candidate key.
CREATE table Department(DeptNo int PRIMARY KEY, DeptName varchar(20) UNIQUE NOT NULL, Location varchar(20)); 

3. Make DNo of Employee as foreign key which refers to DeptNo of Department.  
ALTER TABLE Employee ADD CONSTRAINT fk_DNo FOREIGN KEY (DNo) REFERENCES Department(DeptNo);
